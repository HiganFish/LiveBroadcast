		)

# 后续会修改为模块化 因为过早模块化 会导致每次修改代码后需要rebuild才能更新 故临时使用这种方式


add_executable(LiveBroadcastServer main.cpp ${ALL_SRCS})
target_link_libraries(LiveBroadcastServer mysqlclient)
